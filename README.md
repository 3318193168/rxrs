use rxrs::*;

#[wasm_bindgen]
pub fn greet()
{
    let output = RefCell::new(String::new());

    let (subj, obs) = Rc::new(Subject::<NO, i32>::new()).clones();

    obs.filter(|v:&_| v%2 == 0 ).take(4).map(|v:&_| format!("*{}", v)).sub(
        |v: String| output.borrow_mut().push_str(&v), ()
    );

    for i in 0..10 {
        subj.next(i);
    }

    //"*0*2*4*6"
    log(&*output.borrow());
}
